Conditional Rendering

Welcome back, {{user.name}}!

@if (user.isAdmin) {

Admin panel available

} @else {

User dashboard

} } @else {

List Rendering

@for (item of items; track item.id; let i = $index, isFirst = $first, isLast = $last) {

Item {{i + 1}}: {{item.name}}

@if (isFirst) { First } @if (isLast) { Last }
} @empty {

No items to display

}

Switch Statement

@switch (userRole) { @case ('admin') {

Full access granted

} @case ('moderator') {

Moderate access granted

} @case ('user') {

Basic access granted

} @default {

Access denied

} }
